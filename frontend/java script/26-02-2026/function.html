function in js

1.function decleration

        function fun_name(parameters){
            //code
        }
        fun_name(args);
2.function expression 
        const fun_name=function(parameters){
            //code
        }
         fun_name(args);
3. arrow function 
        const fun_name=function(parameters) =>{
            //code
        }
         fun_name(args);
